<!-- Section -->
<section class="description">

<p>The <dfn>INPUT</dfn> element defines an input field. When you specify "<span class="soValue">submit</span>" (or "<span class="soValue">reset</span>") for the <span class="soAttr">type</span> attribute of this element, a submit button (or a reset button) is created.</p>

</section>
<!-- /Section -->


<!-- Section -->
<section class="code">

<pre class="preBreak">
<code>
<mark>&lt;input <span class="soAttr">type="<span class="soValue">submit</span>" value="<span class="soValue">Submit</span>"</span>&gt;</mark>

<mark>&lt;input <span class="soAttr">type="<span class="soValue">reset</span>" value="<span class="soValue">Reset</span>"</span>&gt;</mark>
</code>
</pre>

</section>
<!-- /Section -->


<!-- Section -->
<section class="description">

<div class="over">
<table>
<tr>
<th>Attribute</th>
<th>Value</th>
<th>Explanation</th>
</tr>
<tr>
<td rowspan="2"><span class="soAttr">type=" "</span></td>
<td><span class="soValue">submit</span></td>
<td rowspan="2">the type of input field<br><span class="soValue">submit</span> : creates a submit button<br><span class="soValue">reset</span> : creates a reset button</td>
</tr>
<tr>
<td><span class="soValue">reset</span></td>
</tr>
<tr>
<td><span class="soAttr">name=" "</span></td>
<td>button name</td>
<td>a unique name for the button</td>
</tr>
<tr>
<td><span class="soAttr">value=" "</span></td>
<td>button text</td>
<td>the text displayed on the button</td>
</tr>
</table>
</div>

<dl>
<dt>type="<span class="soValue">submit</span>"</dt>
<dd>Creates a submit button on the form. When this button is clicked, the form data is submitted to the server.</dd>
<dt>type="<span class="soValue">reset</span>"</dt>
<dd>Creates a reset button on the form. When this button is clicked, the input is reset.</dd>
<dt>name=""</dt>
<dd>The button name is used to identify the clicked submit button.</dd>
<dt>value=""</dt>
<dd>Value is the text displayed on the button.</dd>
</dl>

</section>
<!-- /Section -->
